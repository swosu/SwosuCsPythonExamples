% Section 1
\section{Why Source Control?}
\begin{frame}{The Problem Without It}
  \begin{itemize}
    \item Lost files named `final_final_v2_REALfinal.py`
    \item No rollback when things go sideways
    \item Collaboration chaos — overwriting your partner’s code
  \end{itemize}
\end{frame}

\begin{frame}{The Solution}
  \begin{block}{Version Control Systems (VCS)}
    A version control system tracks changes to your code, lets you collaborate safely, and makes you look organized.
  \end{block}
  \begin{exampleblock}{Examples}
    Git, Mercurial, Subversion
  \end{exampleblock}
\end{frame}

\section{Best Practices}
\begin{frame}{Pro Tips}
  \begin{itemize}
    \item Commit early, commit often.
    \item Use meaningful commit messages.
    \item Branch before you break things.
    \item Pull before you push.
  \end{itemize}
\end{frame}
