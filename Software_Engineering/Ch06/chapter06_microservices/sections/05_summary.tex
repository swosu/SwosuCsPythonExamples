\section{Summary and Further Reading}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Microservices split a system into independently deployable services.
    \item They improve modularity, scalability, and team autonomy.
    \item They also introduce distributed-system complexity.
    \item They work best when supported by automation, monitoring, and strong design practices.
\end{itemize}
\end{frame}

\begin{frame}{Further Reading}
\begin{itemize}
    \item Sommerville, \textit{Software Engineering}
    \item Martin Fowler on Microservices
    \item Sam Newman, \textit{Building Microservices}
\end{itemize}
\end{frame}
