\section{Core Concepts}

\begin{frame}{Key Characteristics of Microservices}
\begin{itemize}
    \item Small and focused services
    \item Independent deployment
    \item Clear service boundaries
    \item Decentralized data and control
    \item Lightweight communication mechanisms
\end{itemize}
\end{frame}

\begin{frame}{Microservices and Business Capabilities}
\begin{itemize}
    \item Services are usually organized around business functions.
    \item Examples:
    \begin{itemize}
        \item User accounts
        \item Orders
        \item Payments
        \item Inventory
    \end{itemize}
    \item This can improve maintainability and team ownership.
\end{itemize}
\end{frame}
